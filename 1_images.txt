            {

                src: "https://images.unsplash.com/photo-1506905925346-21bda4d32df4?ixlib=rb-1.2.1&auto=format&fit=crop&w=500&q=60"
            },
            {
                id: 5,
                title: "Modern Architecture",
                category: "architecture",
                gender: "male",
                src: "https://images.unsplash.com/photo-1487958449943-2429e8be8625?ixlib=rb-1.2.1&auto=format&fit=crop&w=500&q=60"
            },
            {
                id: 6,
                title: "Colorful Abstraction",
                category: "abstract",
                gender: "female",
                src: "https://images.unsplash.com/photo-1543857778-c4a1a569eebe?ixlib=rb-1.2.1&auto=format&fit=crop&w=500&q=60"
            }
        ];

        let images = [...initialImages];
        let nextId = 7;

        // Initialize gallery
        document.addEventListener('DOMContentLoaded', function() {
            renderGallery();
            setupEventListeners();
        });

        function renderGallery(filter = 'all') {
            const gallery = document.getElementById('imageGallery');
            gallery.innerHTML = '';

            const filteredImages = images.filter(image => {
                if (filter === 'all') return true;
                return image.category === filter || image.gender === filter;
            });

            if (filteredImages.length === 0) {
                gallery.innerHTML = '<p style="grid-column: 1 / -1; text-align: center; padding: 2rem;">No images found for this filter.</p>';
                return;
            }

            filteredImages.forEach(image => {
                const galleryItem = document.createElement('div');
                galleryItem.className = 'gallery-item';
                galleryItem.setAttribute('data-category', image.category);
                galleryItem.setAttribute('data-gender', image.gender);

                galleryItem.innerHTML = `
                    <img src="${image.src}" alt="${image.title}" class="gallery-img">
                    <div class="gallery-info">
                        <h3 class="gallery-title">${image.title}</h3>
                        <div class="gallery-meta">
                            <span class="gallery-category category-${image.category}">${image.category}</span>
                            <span class="gallery-gender gender-${image.gender}">${image.gender}</span>
                        </div>
                    </div>
                `;

                gallery.appendChild(galleryItem);
            });
        }

        function setupEventListeners() {
            // Filter buttons
            const filterButtons = document.querySelectorAll('.filter-btn');
            filterButtons.forEach(button => {
                button.addEventListener('click', function() {
                    filterButtons.forEach(btn => btn.classList.remove('active'));
                    this.classList.add('active');
                    const filterValue = this.getAttribute('data-filter');
                    renderGallery(filterValue);
                });
            });

            // Upload form
            const uploadForm = document.getElementById('uploadForm');
            uploadForm.addEventListener('submit', function(e) {
                e.preventDefault();
                
                const title = document.getElementById('imageTitle').value;
                const category = document.getElementById('imageCategory').value;
                const gender = document.getElementById('imageGender').value;
                const fileInput = document.getElementById('imageUpload');
                
                if (!fileInput.files[0]) {
                    alert('Please select an image to upload.');
                    return;
                }

                const file = fileInput.files[0];
                const reader = new FileReader();

                reader.onload = function(e) {
                    const newImage = {
                        id: nextId++,
                        title: title,
                        category: category,
                        gender: gender,
                        src: e.target.result
                    };

                    images.push(newImage);
                    renderGallery();
                    
                    // Reset form
                    uploadForm.reset();
                    
                    alert('Image uploaded successfully!');
                };

                reader.readAsDataURL(file);
            });
        }
